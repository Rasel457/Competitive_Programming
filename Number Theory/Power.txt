ll power(ll num,ll p)
{
    LL ret=0;
    if(p==0)
    {
        return 1LL;
    }
    if(p==1)
    {
        return num;
    }
    if(p%2==0)
    {
        ret=power(num,p/2);
        return ret*ret;
    }
    else{
        ret=power(num,p/2);
        return ret*ret*num;
    }
}

int Pow(int a,int b)
{
    int x;
    if(b==0)
    {
        return 1;
    }
    x=Pow(a,b-1);
    return a*x;
}
