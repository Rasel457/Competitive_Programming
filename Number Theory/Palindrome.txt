#include <bits/stdc++.h>
using namespace std;
typedef long long int LL;

int main()
{
    bool flag=false;
    string s;
    cin>>s;
    int l=0,r=s.size()-1;//5
    int cnt=0;
    for(int i=s.size()-1;i>=0;i--)
    {
        r=i;
        if(s[i]!='a')
        {
            break;
        }
        cnt++;
    }
    for(int i=0;i<s.size()-1;i++)
    {
        l=i;
        if(cnt==0)break;
        if(s[i]!='a')
        {
            break;
        }
        cnt--;
    }
    while(l<=r)
    {
        if(s[l]!=s[r])
        {
            flag=true;
            break;
        }
        l++,r--;
    }

    if(!flag)
    {
        cout<<"Yes\n";
    }
    else
    {
        cout<<"No\n";
    }

    return 0;
